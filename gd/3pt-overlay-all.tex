\documentclass[tikz]{standalone}
\usepackage{pgfplots}
\usepackage{xcolor}
\usetikzlibrary{calc}
\definecolor{brown}{HTML}{8a4e3d}
\definecolor{blue}{HTML}{6d9eeb}
\definecolor{gray}{HTML}{888888}
\definecolor{gold}{HTML}{ffd966}
\pgfplotsset{compat=1.18}

\begin{document}
\begin{tikzpicture}
  \begin{axis}[
    axis lines=left,
    x axis line style={->,>=stealth, shorten >=-3pt},
    y axis line style={->,>=stealth, shorten >=-3pt},
    xlabel={\(\theta\)},
    ylabel style={rotate=-90},
    xmin=0, xmax=4,
    ymin=0, ymax=25,
    xtick={0,1,...,4},
    ytick={0,5,...,25},
    minor x tick num=1,
    minor y tick num=1,
    grid=both,
    major grid style={line width=0.2pt,draw=black},
    minor grid style={line width=0.1pt,draw=gray!50,dashed},
    width=10cm,
    height=10cm,
  ]
    % Individual curves
    \addplot[domain=0:4, samples=200, thick, color=brown!70]
      {(2*x-5)^2};

    \addplot[domain=0:4, samples=200, thick, color=blue!70]
      {(3*x-6)^2};

    \addplot[domain=0:4, samples=200, thick, color=gray!70]
      {(4*x-7)^2};

    % Combined J(θ)
    \addplot[domain=0:4, samples=200, ultra thick, color=gold]
      {((2*x-5)^2 + (3*x-6)^2 + (4*x-7)^2)/3};

    % Tangent lines at θ=2.5
    % J1: slope 0, value 0
    \addplot[domain=1:4, samples=2, thick, color=brown!50, dashed]
      {0};

    % J2: slope 9, value 2.25
    \addplot[domain=1:4, samples=2, thick, color=blue!50, dashed]
      {9*(x-2.5)+2.25};

    % J3: slope 24, value 9
    \addplot[domain=1.5:3.167, samples=2, thick, color=gray!60, dashed]
      {24*(x-2.5)+9};

    % Combined tangent: slope 11, value J(2.5) = (0+2.25+9)/3 = 3.75
    \addplot[domain=1:4, samples=2, thick, color=gold!80!black, dashed]
      {11*(x-2.5)+3.75};

    % Diamond markers at θ=2.5
    \addplot[only marks, mark=diamond*, mark options={scale=3, fill=brown}]
      coordinates {(2.5,0)};
    \addplot[only marks, mark=diamond*, mark options={scale=3, fill=blue}]
      coordinates {(2.5,2.25)};
    \addplot[only marks, mark=diamond*, mark options={scale=3, fill=gray}]
      coordinates {(2.5,9)};
    \addplot[only marks, mark=diamond*, mark options={scale=3, fill=gold}]
      coordinates {(2.5,3.75)};

  \end{axis}
\end{tikzpicture}
\end{document}
