\documentclass[tikz]{standalone}
\usepackage{pgfplots}
\usepackage{xcolor} % For using hex colors
\usetikzlibrary{calc}
\definecolor{mycolor}{HTML}{ffd966} % Define custom color #ffd966
\pgfplotsset{compat=1.18}

\begin{document}
\begin{tikzpicture}
  \begin{axis}[
    % Draw only the bottom and left axes with arrow tips:
    axis lines=left,
    x axis line style={->,>=stealth, shorten >=-3pt},
    y axis line style={->,>=stealth, shorten >=-3pt},
    xlabel={\(\theta\)},
    ylabel={\(J(\theta)\)},
    % Rotate the y label so that it appears upside down:
    ylabel style={rotate=-90},
    xmin=0, xmax=4,
    ymin=0, ymax=5,
    % Major ticks at integer values...
    xtick={0,1,...,3},
    ytick={0,1,...,4},
    % ...and an extra tick at 4 (without grid lines):
    extra x ticks={4},
    extra x tick style={grid=none},
    extra y ticks={5},
    extra y tick style={grid=none},
    % One minor tick between each major tick:
    minor x tick num=1,
    minor y tick num=1,
    grid=both,
    % Grid styling:
    major grid style={line width=0.2pt,draw=black},
    minor grid style={line width=0.1pt,draw=gray!50,dashed},
    width=10cm,
    height=10cm,
  ]
    % Plot the constant function J(θ) = (1+4+9)/3 = 14/3 ≈ 4.67
    \addplot[domain=0:4, samples=200, ultra thick, color=mycolor] {14/3};
    
    % Add a label to show the function value
    \node[above] at (axis cs:2,3) {\(J = \frac{1^2+2^2+3^2}{3}\)};
  \end{axis}
\end{tikzpicture}
\end{document}
