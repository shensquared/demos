    ylabel={\(J(\theta)\)},
    ylabel style={rotate=-90},
    xmin=0, xmax=4.5,
    ymin=0, ymax=4.5,

    ytick={0,1,...,4},
    minor x tick num=1,
    minor y tick num=1,
    grid=both,

    minor grid style={line width=0.1pt,draw=gray!50,dashed},

      {9*(x - 2)^2};
